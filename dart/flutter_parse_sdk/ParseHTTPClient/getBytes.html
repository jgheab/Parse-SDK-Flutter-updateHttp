  String path, {

  dynamic cancelToken,

  final http.Response response = await _client.get(
    Uri.parse(path),
    headers: options?.headers,
  );
  return ParseNetworkByteResponse(
      bytes: response.bodyBytes, statusCode: response.statusCode);
